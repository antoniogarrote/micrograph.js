  4 /**
  5  * Creates a new persistent Micrograph instance.
  6  *
  7  * @param {String} [name] a name identifying the persistent graph.
  8  * @param {bool} [overwrite] flag indicating if the data in the graph must be overwritten.
  9  * @param {Object} [options] a hash of options for the graph constructor. See Micrograph function.
 10  * @param {Function} [callback] optional callback function that will be invoked with the new graph instance.
 11  */
 12 mg.open = function(name,overwrite,options,callback) {
 13     if(!callback) {
 14 	if(typeof(options) === 'function') {
 15 	    callback = options;
 16 	    options = {};
 17 	}  else {
 18 	    throw "Persistent storage requires a callback functon";
 19 	}
 20     }
 21     options['persistent'] = true;
 22     options['name'] = name;
 23     options['overwrite'] = overwrite;
 24     
 25     if(overwrite && localStorage && localStorage.clear)
 26 	localStorage.clear()
 27 
 28     new mg(options, callback);
 29 };
